Write a Python function to find prime numbers
def primes(n):
  return [i for i in range(2, n) if all(i % j != 0 for j in range(2, int(i**0.5)+1))]
Now make it async and add caching
import functools

@functools.lru_cache(maxsize=128)
def primes(n):
  return [i for i in range(2, n) if all(i % j != 0 for j in range(2, int(i**0.5)+1))]
